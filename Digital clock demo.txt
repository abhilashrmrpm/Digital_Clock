    @Override
    public void run() {
        while (true) {
            Calendar cal = Calendar.getInstance();
            //get current hour minute and second 
            hour = cal.get(Calendar.HOUR_OF_DAY);
            minute = cal.get(Calendar.MINUTE);
            second = cal.get(Calendar.SECOND);
            
            //for 24 hours time 
            SimpleDateFormat sdf24 = new SimpleDateFormat("HH:mm:SS");
            Date dat = cal.getTime();
            String time24 = sdf24.format(dat);
            
            //set to label 24hour time
            TimeIn24.setText(time24);   //TimeIn24 is a label
            
            //for 12 hours 
            SimpleDateFormat sdf12 = new SimpleDateFormat("hh:mm:SS aa"); //aa means AM and PM in Time
            String time12 = sdf12.format(dat);
            
            //set to label 24 hour time 
            TimeIn12.setText(time12);  //TimeIn12 is a label
            
            
            
            
            
        }
    }
